    <script>
        // =========================================================================
        // URL ของ Google Apps Script Web App (ต้องนำมาใส่หลังจาก Deploy Apps Script)
        // =========================================================================
        const GOOGLE_SCRIPT_URL = 'https://script.google.com/macros/s/AKfycbwYCRtiWdE6cUIYMdBh1kUAS8aXyIFGeUJe2MEsxbgpqnSILKYqy080o5QwgkDO0rbyMg/exec';

        window.handleFile = function(input) {
            if (input.files && input.files[0]) {
                document.getElementById('uploadPlaceholder').classList.add('hidden');
                document.getElementById('filePreview').classList.remove('hidden');
            }
        };

        // ฟังก์ชันแปลงรูปภาพเป็น Base64 เพื่อส่งผ่าน HTTP Request
        function getBase64(file) {
            return new Promise((resolve, reject) => {
                const reader = new FileReader();
                reader.readAsDataURL(file);
                reader.onload = () => {
                    // ตัดส่วน 'data:image/jpeg;base64,' ออก เอาแค่ข้อมูล
                    resolve(reader.result.split(',')[1]); 
                };
                reader.onerror = error => reject(error);
            });
        }

        window.submitForm = async function() {
            const firstName = document.getElementById('firstName').value.trim();
            const lastName = document.getElementById('lastName').value.trim();
            const studentId = document.getElementById('studentId').value.trim();
            const phoneNumber = document.getElementById('phoneNumber').value.trim();
            const prefix = document.getElementById('platePrefix').value.trim();
            const number = document.getElementById('plateNumber').value.trim();
            const province = document.getElementById('plateProvince').value;
            const fileInput = document.getElementById('fileInput');
            const submitBtn = document.getElementById('submitBtn');
            const errorMsg = document.getElementById('errorMsg');

            if (!firstName || !lastName || !studentId || !phoneNumber || !prefix || !number || !province) {
                showError("กรุณากรอกข้อมูลให้ครบถ้วน");
                return;
            }

            if (fileInput.files.length === 0) {
                showError("กรุณาอัปโหลดรูปภาพรถจักรยานยนต์");
                return;
            }

            if (GOOGLE_SCRIPT_URL === 'ใส่_URL_WEB_APP_ที่ได้จาก_Google_Apps_Script_ที่นี่') {
                showError("⚠️ ระบบยังไม่ได้เชื่อมต่อ Google Apps Script (กรุณาดูคู่มือตั้งค่า)");
                return;
            }

            submitBtn.disabled = true;
            submitBtn.innerHTML = `<span class="loader"></span> <span class="ml-2">กำลังอัปโหลดข้อมูล...</span>`;
            errorMsg.classList.add('hidden');

            try {
                // 1. แปลงไฟล์ภาพเป็น Base64
                const file = fileInput.files[0];
                const base64Data = await getBase64(file);
                
                // 2. เตรียมข้อมูล Payload
                const payload = {
                    firstName: firstName,
                    lastName: lastName,
                    studentId: studentId,
                    phoneNumber: phoneNumber,
                    plate: `${prefix}-${number}`,
                    province: province,
                    imageName: `Moto_${studentId}_${Date.now()}`,
                    mimeType: file.type,
                    imageData: base64Data
                };

                // 3. ส่งข้อมูลไปยัง Google Apps Script
                const response = await fetch(GOOGLE_SCRIPT_URL, {
                    method: 'POST',
                    // No 'Content-Type' header to avoid CORS preflight issues with Google Script
                    body: JSON.stringify(payload) 
                });

                const result = await response.json();

                if (result.status === 'success') {
                    // แสดงผลหน้าสำเร็จ
                    document.getElementById('welcomeUser').innerText = `นักศึกษา: ${firstName} ${lastName}`;
                    document.getElementById('plateLabel').innerText = `${prefix.toUpperCase()}-${number}`;
                    document.getElementById('provinceLabel').innerText = province;
                    
                    // สร้าง QR Code
                    document.getElementById('qrImage').src = `https://api.qrserver.com/v1/create-qr-code/?size=200x200&data=BUMS-STU-${studentId}`;
                    
                    document.getElementById('regForm').classList.add('hidden');
                    document.getElementById('successMessage').classList.remove('hidden');
                } else {
                    throw new Error(result.message || "เกิดข้อผิดพลาดจากเซิร์ฟเวอร์");
                }
                
            } catch (error) {
                console.error("Submission error:", error);
                showError("เกิดข้อผิดพลาด: " + error.message);
            } finally {
                submitBtn.disabled = false;
                submitBtn.innerHTML = `<span>ลงทะเบียน</span>`;
            }
        };

        window.resetForm = function() {
            document.getElementById('regForm').reset();
            document.getElementById('uploadPlaceholder').classList.remove('hidden');
            document.getElementById('filePreview').classList.add('hidden');
            document.getElementById('regForm').classList.remove('hidden');
            document.getElementById('successMessage').classList.add('hidden');
            document.getElementById('errorMsg').classList.add('hidden');
        };

        function showError(msg) {
            const errorMsg = document.getElementById('errorMsg');
            errorMsg.innerText = msg;
            errorMsg.classList.remove('hidden');
        }
    </script>
